Apps Script Code

Paste this into Apps Script and re-deploy. It writes to a named "Sessions" tab, stores a hidden UID per row (so repeated syncs can't duplicate), and adds a getAll action so the owner dashboard can show every device's data. Same code lives in apps-script.gs in the project.
var SHEET_ID = '1bJXuhnfsXKMCxBKwf7wVBKzSRXiNPQdFPdn2_8lkDGk'; var VISIT_HEADERS = ['Console','Date','Start Time','End Time','Sessions', 'Minutes','Amount','Payment','Split Detail','Overtime','Collected At','UID']; var SALE_HEADERS = ['Date','Time','Item','Qty','Amount','Payment','UID']; function doGet(e) { try { var ss = SpreadsheetApp.openById(SHEET_ID); var action = e.parameter.action; if (action === 'getPin') { var cell = e.parameter.cell || 'Config!A1'; return out({pin: String(ss.getRange(cell).getValue())}); } if (action === 'getAll') { return out({ok:true, visits: readSheet(ss.getSheetByName('Sessions')), sales: readSheet(ss.getSheetByName('Sales'))}); } if (action === 'sync') { var data = JSON.parse(decodeURIComponent(e.parameter.data)); var type = data.type; // 'visit' or 'sale' var sheet, headers; if (type === 'sale') { sheet = ss.getSheetByName('Sales') || ss.insertSheet('Sales'); headers = SALE_HEADERS; } else { sheet = ss.getSheetByName('Sessions') || ss.insertSheet('Sessions'); headers = VISIT_HEADERS; } if (sheet.getLastRow() === 0) sheet.appendRow(headers); // Collect UIDs already stored (last column) so we never append a duplicate. var uidCol = headers.length; var seen = {}; if (sheet.getLastRow() > 1) { var stored = sheet.getRange(2, uidCol, sheet.getLastRow() - 1, 1).getValues(); for (var i = 0; i < stored.length; i++) seen[String(stored[i][0])] = true; } data.rows.forEach(function (r) { if (r.id && seen[r.id]) return; // already saved — skip duplicate if (type === 'sale') { sheet.appendRow([r.date, r.time, r.item, r.qty, r.amount, r.payment, r.id || '']); } else { sheet.appendRow(['PS ' + r.psId, r.date, r.startTime, r.endTime, r.sessions, r.sessions * (r.duration || 12), r.amount, r.paymentMethod, r.splitDetail || '', r.overtime || '', r.collectedAt, r.id || '']); } if (r.id) seen[r.id] = true; }); return out({ok:true}); } return out({ok:false,error:'unknown action'}); } catch(err) { return out({ok:false,error:err.toString()}); } } function readSheet(sheet) { if (!sheet || sheet.getLastRow() < 2) return []; var values = sheet.getDataRange().getDisplayValues(); var headers = values[0], rows = []; for (var i = 1; i < values.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) obj[headers[j]] = values[i][j]; rows.push(obj); } return rows; } function out(obj){ return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
