Apps Script Code

function doGet(e) { try { var ss = SpreadsheetApp.openById('1bJXuhnfsXKMCxBKwf7wVBKzSRXiNPQdFPdn2_8lkDGk'); var action = e.parameter.action; if (action === 'getPin') { var cell = e.parameter.cell || 'Config!A1'; var val = ss.getRange(cell).getValue(); return out({pin: String(val)}); } if (action === 'sync') { var data = JSON.parse(decodeURIComponent(e.parameter.data)); var type = data.type; // 'visit' or 'sale' var sheet; if (type === 'sale') { sheet = ss.getSheetByName('Sales') || ss.insertSheet('Sales'); if (sheet.getLastRow()===0) sheet.appendRow(['Date','Time','Item','Qty','Amount','Payment']); data.rows.forEach(function(r){ sheet.appendRow([r.date,r.time,r.item,r.qty,r.amount,r.payment]); }); } else { sheet = ss.getActiveSheet(); if (sheet.getLastRow()===0) sheet.appendRow(['Console','Date','Time','Sessions','Minutes','Amount','Payment','Collected At']); data.rows.forEach(function(r){ sheet.appendRow(['PS '+r.psId,r.date,r.startTime,r.sessions,r.sessions*7,r.amount,r.paymentMethod,r.collectedAt]); }); } return out({ok:true}); } return out({ok:false,error:'unknown action'}); } catch(err) { return out({ok:false,error:err.toString()}); } } function out(obj){ return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
